h2. Basic Ruby Syntax

h3. Sub heading

h3. Sub heading

<% coderay(:lang => "ruby") do -%>
def hello_world(name)
  puts "Hello #{name}!"

#=> nil

#=> Hello World!
#=> nil
<% end -%>

h3. Sub heading

<% coderay(:lang => "ruby") do -%>
def hello_world(name = "World")
  puts "Hello #{name.capitalize}!"
end
#=> nil
hello_world "odin"
#=> Hello Odin!
#=> nil
hello_world
#=> Hello World!
#=> nil
<% end -%>

h3. Sub heading

<% coderay(:lang => "ruby") do -%>
class Greeter
  def initialize(name = "World")
    @name = name
  end
  def say_hi
    puts "Hi #{@name}!"
  end
  def say_bye
    puts "Bye #{@name}, come back soon."
  end
end
#=> nil
<% end -%>
